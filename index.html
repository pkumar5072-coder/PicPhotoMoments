    <script>

        // --- 1. SETTINGS (Sirf yahan number badlein) ---

        const config = {

            wedding: 5,  // Agar 100 photo hain to yahan 100 likhein

            pre: 3,      // Pre-wedding ki photos ka number

            eng: 2,      // Engagement photos ka number

            haldi: 2,

            meh: 2

        };


        // --- 2. VIDEOS (Yahan YouTube ID daalein) ---

        const videos = [

            { id: "YOUR_YOUTUBE_ID", cat: "wedding" },

            { id: "YOUR_YOUTUBE_ID", cat: "pre" }

        ];


        const gallery = document.getElementById('gallery');


        function loadGallery() {

            gallery.innerHTML = "";

            // Load Photos Automatically

            for (let cat in config) {

                for (let i = 1; i <= config[cat]; i++) {

                    gallery.innerHTML += `<div class="gallery-item ${cat} photo"><img src="${cat}${i}.jpg" onerror="this.parentElement.style.display='none'"></div>`;

                }

            }

            // Load Videos

            videos.forEach(v => {

                gallery.innerHTML += `<div class="gallery-item ${v.cat} video"><iframe src="https://www.youtube.com/embed/${v.id}"></iframe></div>`;

            });

        }


        loadGallery();


        // Filtering Logic

        const catBtns = document.querySelectorAll('#cat-filters .filter-btn');

        const typeBtns = document.querySelectorAll('#type-filters .filter-btn');

        const subF = document.getElementById('type-filters');


        function filter() {

            const c = document.querySelector('#cat-filters .active').dataset.filter;

            const t = document.querySelector('#type-filters .active').dataset.type;

            document.querySelectorAll('.gallery-item').forEach(item => {

                const cm = c === 'all' || item.classList.contains(c);

                const tm = t === 'all-t' || (t === 'photo' && item.classList.contains('photo')) || (t === 'video' && item.classList.contains('video'));

                item.style.display = (cm && tm) ? "block" : "none";

            });

        }


        catBtns.forEach(b => b.addEventListener('click', () => {

            document.querySelector('#cat-filters .active').classList.remove('active');

            b.classList.add('active');

            subF.style.display = (b.dataset.filter === 'all') ? "none" : "flex";

            filter();

        }));


        typeBtns.forEach(b => b.addEventListener('click', () => {

            document.querySelector('#type-filters .active').classList.remove('active');

            b.classList.add('active');

            filter();

        }));

    </script>
